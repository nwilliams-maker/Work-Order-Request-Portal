async function submitToFormspree() { const decision = document.querySelector('input[name="ic_decision"]:checked').value; const sig = document.getElementById('typed_sig').value; const consent = document.getElementById('legal_consent').checked; const counter = document.getElementById('ic_counter_val').value; const notes = document.getElementById('ic_notes').value; if (decision === 'Accepted' && (!sig || !consent)) { return alert("Signature and consent required to accept."); } const btn = document.getElementById('submitBtn'); btn.disabled = true; btn.innerHTML = " Saving Response..."; const rawJobContent = window.payloadData.jobOnly ? window.payloadData.jobOnly.split('|') : []; const locationList = rawJobContent.map((loc, index) => `${index + 1}. ${loc}`).join('\n'); let emailDate = 'N/A'; if (window.payloadData.due) { const parts = window.payloadData.due.split('-'); emailDate = parts.length === 3 ? `${parts[1]}/${parts[2]}/${parts[0]}` : window.payloadData.due; } const formData = { "Decision": decision, "Contractor": window.payloadData.icn, "Work Order": window.payloadData.wo, "Due Date": emailDate, "Locations": window.payloadData.lCnt, "Tasks": window.payloadData.tCnt, "Kiosks": window.payloadData.kCnt, "Digital": window.payloadData.dCnt || 0, "Mileage": window.payloadData.mi, "Original Comp": `$${window.payloadData.comp}`, "Counter Offer": counter ? `$${counter}` : "N/A", "Dispatcher Notes": window.payloadData.notes || "None", "IC Notes": notes || "None", "Signature": sig || "N/A", "TASK LOCATION LIST": "\n" + locationList, "_subject": `${decision}: WO #${window.payloadData.wo} (${window.payloadData.icn})` }; try { const response = await fetch(formspreeUrl, { method: 'POST', headers: { 'Content-Type': 'application/json', 'Accept': 'application/json' }, body: JSON.stringify(formData) }); if (response.ok) { if (decision === 'Accepted') { if (!window.payloadData.phone) { alert("⚠️ Route accepted, but the contractor is missing a phone number. Tasks were NOT auto-assigned in Onfleet."); } else if (!window.payloadData.taskIds || window.payloadData.taskIds.trim() === '') { alert("⚠️ Route accepted, but no tasks were attached. Auto-assign skipped."); } else { btn.innerHTML = " Assigning Tasks in Onfleet..."; const assignUrl = `${webAppUrl}?action=assignTasks&phone=${encodeURIComponent(window.payloadData.phone)}&taskIds=${encodeURIComponent(window.payloadData.taskIds)}`; try { const gasRes = await fetch(assignUrl); const data = await gasRes.json(); if (data.success) { console.log(`✅ Onfleet Success! ${data.msg}`); } else { alert(`❌ Auto-Assign Failed: ${data.error}`); } } catch (e) { alert("❌ Auto-assign network request failed to reach Google."); } } if (window.currentRouteId) { try { // PASSES INFO TO GOOGLE TO EMAIL THE DISPATCHER fetch(`${webAppUrl}?action=acceptRoute&routeId=${encodeURIComponent(window.currentRouteId)}`); } catch(e) {} } } else if (decision === 'Declined') { if (window.currentRouteId) { try { // PASSES INFO TO GOOGLE TO EMAIL THE DISPATCHER fetch(`${webAppUrl}?action=declineRoute&routeId=${encodeURIComponent(window.currentRouteId)}¬es=${encodeURIComponent(notes || 'None')}&counter=${encodeURIComponent(counter || 'N/A')}`); } catch(e) {} } } document.getElementById('icMode').classList.add('hidden'); document.getElementById('successView').classList.remove('hidden'); } else { const errText = await response.text(); throw new Error(errText || "Formspree rejected the request."); } } catch (e) { console.error(e); alert("Submission failed to send to Formspree. Error: " + e.message); btn.disabled = false; btn.innerText = "Submit Final Response"; } }
